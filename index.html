Quick pipeline snippet

# Minimal Jenkins Declarative Pipeline
pipeline {
    agent any
    stages {
        stage('Checkout') { steps { checkout scm } }
        stage('Build')    { steps { sh 'echo build...' } }
        stage('Test')     { steps { sh 'echo test...' } }
        stage('Publish')  { steps { sh 'echo publish image...' } }
    }
}
